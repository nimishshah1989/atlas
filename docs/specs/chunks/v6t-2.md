## Background

`backend/services/tv/bridge.py` currently wraps an HTTP client that calls a local
Node.js MCP sidecar at `http://127.0.0.1:7100`. V6T-1 pinned
`tradingview-screener==3.1.0` which lets us query TradingView directly from Python.

The library API (confirmed working):

```python
from tradingview_screener import Query, col

# Fetch TA + screener data for a single NSE stock
_, df = (
    Query()
    .select(
        'name',
        'Recommend.All',    # composite recommendation: -1 to +1
        'Recommend.MA',     # moving average recommendation
        'Recommend.Other',  # oscillator recommendation
        'RSI',              # RSI(14) current
        'RSI[1]',           # RSI(14) previous bar
        'MACD.macd',
        'MACD.signal',
        'EMA200',
        'SMA20',
        'BB.upper',
        'BB.lower',
        'close',
        'volume',
    )
    .set_tickers(f'NSE:{symbol}')
    .set_markets('india')
).get_scanner_data()

row = df.iloc[0].to_dict()  # single row for the symbol
```

- `Recommend.All` range: -1.0 (STRONG_SELL) to +1.0 (STRONG_BUY)
- Thresholds: ≥0.5 BUY, 0.1–0.49 WEAK_BUY, -0.09–0.09 NEUTRAL,
  -0.49–-0.1 WEAK_SELL, ≤-0.5 SELL
- If symbol not found: df is empty (len == 0) → raise TVBridgeUnavailableError

### 1. `backend/services/tv/bridge.py` — REWRITE

Replace the HTTP sidecar client with the tradingview_screener library.

```python
"""TV bridge — direct tradingview_screener library integration.

Previously wrapped an HTTP sidecar; now queries TradingView in-process.
TVBridgeUnavailableError is raised when the TradingView API is unreachable
or the symbol is not found, so callers remain unchanged.
"""
from __future__ import annotations

from typing import Any

import structlog

log = structlog.get_logger(__name__)


class TVBridgeUnavailableError(Exception):
    """Raised when TradingView data cannot be fetched."""


# TA columns fetched for every symbol
_TA_COLUMNS = [
    "name",
    "Recommend.All",
    "Recommend.MA",
    "Recommend.Other",
    "RSI",
    "RSI[1]",
    "MACD.macd",
    "MACD.signal",
    "EMA200",
    "SMA20",
    "BB.upper",
    "BB.lower",
    "close",
    "volume",
]

_SCREENER_COLUMNS = [
    "name",
    "close",
    "volume",
    "market_cap_basic",
    "price_earnings_ttm",
    "earnings_per_share_basic_ttm",
    "52_week_high",
    "52_week_low",
]

_FUNDAMENTAL_COLUMNS = [
    "name",
    "earnings_per_share_basic_ttm",
    "price_earnings_ttm",
    "price_to_book_ratio",
    "debt_to_equity",
    "return_on_equity",
    "gross_margin",
    "net_margin",
    "revenue_per_employee",
    "market_cap_basic",
]


class TVBridgeClient:
    """In-process TradingView data client using tradingview_screener library."""

    async def get_ta_summary(
        self,
        symbol: str,
        exchange: str,
        interval: str,
    ) -> dict[str, Any]:
        """Fetch TA summary for a symbol.

        Args:
            symbol: Ticker (e.g. 'RELIANCE').
            exchange: Exchange code (e.g. 'NSE') — used to prefix ticker.
            interval: Ignored in library mode (library returns daily data).

        Returns:
            Dict with TA fields.

        Raises:
            TVBridgeUnavailableError: If symbol not found or network error.
        """
        return await self._fetch(symbol, exchange, _TA_COLUMNS)

    async def get_screener(
        self,
        symbol: str,
        exchange: str,
    ) -> dict[str, Any]:
        return await self._fetch(symbol, exchange, _SCREENER_COLUMNS)

    async def get_fundamentals(
        self,
        symbol: str,
        exchange: str,
    ) -> dict[str, Any]:
        return await self._fetch(symbol, exchange, _FUNDAMENTAL_COLUMNS)

    async def _fetch(
        self,
        symbol: str,
        exchange: str,
        columns: list[str],
    ) -> dict[str, Any]:
        try:
            from tradingview_screener import Query  # lazy import
            ticker = f"{exchange}:{symbol}"
            _, df = (
                Query()
                .select(*columns)
                .set_tickers(ticker)
                .set_markets("india")
            ).get_scanner_data()

            if df.empty:
                raise TVBridgeUnavailableError(f"Symbol {ticker!r} not found in TradingView")

            row: dict[str, Any] = df.iloc[0].to_dict()
            log.debug("tv_bridge_fetch_ok", symbol=symbol, columns=len(columns))
            return row
        except TVBridgeUnavailableError:
            raise
        except Exception as exc:
            log.warning("tv_bridge_fetch_error", symbol=symbol, error=str(exc))
            raise TVBridgeUnavailableError(f"TradingView fetch failed for {symbol}: {exc}") from exc
```

## Important Notes

- `get_scanner_data()` is **synchronous** — run it in `asyncio.get_event_loop().run_in_executor(None, ...)` or use `asyncio.to_thread()` to avoid blocking the event loop
- The library queries `https://scanner.tradingview.com/india/scan` — this is a real HTTP call to TradingView's API
- Exchange prefix for NSE stocks: `NSE:SYMBOL`
- `TVBridgeUnavailableError` keeps the same name so all callers (`tv.py`, `stocks.py`, `watchlists.py`) need zero changes
- `TVCacheService` wraps `TVBridgeClient` and caches results — it stays unchanged
- The `interval` parameter is ignored in library mode; document this clearly
- Tests must mock `tradingview_screener.Query` to avoid real network calls
